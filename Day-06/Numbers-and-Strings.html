outliers <- function(x) {
    a <- quantile(x)[2]
    b <- quantile(x)[4]
    as.logical(x < a | x > b)

}

lettersMatch <- function(words, pattern) {
    a = grep(pattern, words)
    return(a)
}
lettersMatch(c("apple", "berry", "carrie", "rr"), "rr")

piseries <- function(k) {
    signs <- -(-1)^c(1:k)
    num <- (seq(1, by = 2, length.out = k) * signs)^(-1)
    sum(4 * num)

}

piseries <- function(k) {
    signs <- -(-1)^c(1:k)
    num <- (seq(1, by = 2, length.out = k) * signs)^(-1)
    sum(4 * num)

}
howClosetoPi <- function(n) {
    ests <- c()
    for (k in 1:n) {
        ests <- c(ests, piseries(k))
    }
    distancefrompi <- ests - pi
    plot(distancefrompi)
}
howClosetoPi(50)


randApproxtoPi <- function(n) {
    x <- runif(n)
    y <- runif(n)
    sd <- x^2 + y^2
    4 * (length(sd[sd < 1])/length(sd))
}
randApproxtoPi(1000)
